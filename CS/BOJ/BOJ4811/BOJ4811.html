#include 

using namespace std; 

int main(){
    long long arr[1001]; 
    long long Catalan[31]; 
    Catalan[0] = 1;
    Catalan[1] = 1;
    Catalan[2] = 2; 
    Catalan[3] = 5;
    Catalan[4] = 14;

    bool end = true; 
    int index = 0; 

    while(end){
        cin >> arr[index];
        if(arr[index] == 0){
            end = false; 
        }
        index++;
    }

    index = 0; 

    for(int i = 5; i <= 30; i++){
        for(int j = 0; j <= i - 1; j++){
            Catalan[i] += Catalan[j] * Catalan[i - j - 1]; 
        }
    }

    while(arr[index] != 0){
        cout << Catalan[arr[index]] << "\n";
        index++; 
    }

    return 0; 
}
